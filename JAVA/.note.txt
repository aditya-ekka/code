   printf { //format and print
      method used to format output
      % [flag] [width] [.precision] [specifier_character]
      
      specifier_character {
         %s, %c, %d, %f, %b
      }.precision {
         %.1f to display upto one digit after decimal
      }flag {
         +     %+f   output a plus for no. > 0
         ,     %,f   comma grouping seperator
         (     %(f   negative numbers are enclosed in ()
         space % f   display a minus for negative, space for positive
      }width {
         %04f     append extra 0 in begining of number if needed to make string width = 4
         %4f      right justified padding with spaces
         %-4f     left justified padding with spaces 
      }
   }

   {
      void hunt();
   }

   public class Rabbit implements Prey
   {
      //@Override
   }
   public class Hawk implements Predator
   {
      //@Override
   }
   public class Fish impleents Prey, Predator   //

## TIMER and TimerTask

   Timer : class that schedules task at specific times or periodically
      used for- sending notifications, scheduled updates, repetive actions

   TimerTask : represents the task that will be executed by the Timer
      we will extend the TimerTask class to define our task
      (create a subclass of TimerTask and @Override run() function)

   Timer timer = new Timer();
   TimerTask task = new TimerTask(){
      @Override
      public void run() {

         while(true){
            LocalTime time = LocalTime.now();
            if (time.equals(LocalTime.of(12,8,0))){
               System.out.println("Happy New Year");
               return;
            }
            System.out.println(time);
         }

      }
   };
   timer.schedule(task, 2000,(int)(Math.PI));

## HASH MAP
   a data str that stores key-value pair 
   keys are unique, but values can be duplicated 
   doesn't maintain any order, but is memorey efficent
   
   import java.util.HashMap;
   {
      HashMap <String, Double> map = new HashMap<>(); //declaration

      map.put ("apple", 5.0);
      map.put ("orange" , 42.4);
      map.put ("banana", 24.4);

      map.put("orange", 12.2);
      //can have same key, but with different value;
      //this overwrites previous value

      System.out.println(map);
      //{orange=12.2, banana=24.4, apple=5.0}
      
      map.remove("apple");
      System.out.println(map.get("orange")); //value
      System.out.println(map.containsKey("banana"));
      System.out.println(map.containsValue(5.0));
      // don't use 5 instead of 5.0
      System.out.println(map.size());

      System.out.println(map.keySet());


      for (String key: map.keySet()) {
         System.out.print(key + " ");
      }
   }
