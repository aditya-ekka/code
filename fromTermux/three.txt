DELIMITER //

CREATE FUNCTION get_grade(p_score INT)
RETURNS VARCHAR(10)
DETERMINISTIC
BEGIN
    IF p_score IS NULL OR p_score NOT BETWEEN 0 AND 100 THEN
        RETURN 'INVALID';
    END IF;

    IF p_score >= 90 THEN RETURN 'A';
    ELSEIF p_score >= 75 THEN RETURN 'B';
    ELSEIF p_score >= 60 THEN RETURN 'C';
    ELSEIF p_score >= 40 THEN RETURN 'D';
    ELSE RETURN 'F';
    END IF;
END //

DELIMITER ;
