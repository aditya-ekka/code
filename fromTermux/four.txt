CREATE FUNCTION second_max(p_numbers JSON)
RETURNS INT
DETERMINISTIC
BEGIN
    DECLARE i INT DEFAULT 0;
    DECLARE n INT;
    DECLARE val INT;
    DECLARE v_max INT;
    DECLARE v_second INT DEFAULT NULL;

    SET n = JSON_LENGTH(p_numbers);

    IF n < 2 THEN
        RETURN NULL;
    END IF;

    SET v_max = JSON_EXTRACT(p_numbers, '$[0]');

    WHILE i < n DO
        SET val = JSON_EXTRACT(p_numbers, CONCAT('$[', i, ']'));
        IF val > v_max THEN
            SET v_max = val;
        END IF;
        SET i = i + 1;
    END WHILE;

    SET i = 0;

    WHILE i < n DO
        SET val = JSON_EXTRACT(p_numbers, CONCAT('$[', i, ']'));

        IF val != v_max THEN
            IF v_second IS NULL OR val > v_second THEN
                SET v_second = val;
            END IF;
        END IF;

        SET i = i + 1;
    END WHILE;

    RETURN v_second;
END //

DELIMITER ;
