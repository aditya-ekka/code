CREATE FUNCTION count_even_above_avg(p_numbers JSON)
RETURNS INT
DETERMINISTIC
BEGIN
    DECLARE i INT DEFAULT 0;
    DECLARE n INT;
    DECLARE val INT;
    DECLARE v_sum INT DEFAULT 0;
    DECLARE v_count INT DEFAULT 0;
    DECLARE v_avg DOUBLE;
    DECLARE result INT DEFAULT 0;

    SET n = JSON_LENGTH(p_numbers);

    IF n = 0 THEN
        RETURN 0;
    END IF;

    -- First pass
    WHILE i < n DO
        SET val = JSON_EXTRACT(p_numbers, CONCAT('$[', i, ']'));
        SET v_sum = v_sum + val;
        SET v_count = v_count + 1;
        SET i = i + 1;
    END WHILE;

    SET v_avg = v_sum / v_count;

    SET i = 0;

    -- Second pass
    WHILE i < n DO
        SET val = JSON_EXTRACT(p_numbers, CONCAT('$[', i, ']'));

        IF val > v_avg AND MOD(val, 2) = 0 THEN
            SET result = result + 1;
        END IF;

        SET i = i + 1;
    END WHILE;

    RETURN result;
END //

DELIMITER ;
