CREATE FUNCTION sum_avg_positive(p_numbers JSON)
RETURNS VARCHAR(100)
DETERMINISTIC
BEGIN
    DECLARE i INT DEFAULT 0;
    DECLARE n INT;
    DECLARE v_sum INT DEFAULT 0;
    DECLARE v_count INT DEFAULT 0;
    DECLARE val INT;

    SET n = JSON_LENGTH(p_numbers);

    IF n = 0 OR p_numbers IS NULL THEN
        RETURN 'Collection is empty';
    END IF;

    WHILE i < n DO
        SET val = JSON_EXTRACT(p_numbers, CONCAT('$[', i, ']'));

        IF val > 0 THEN
            SET v_sum = v_sum + val;
            SET v_count = v_count + 1;
        END IF;

        SET i = i + 1;
    END WHILE;

    IF v_count = 0 THEN
        RETURN 'No positive numbers found';
    END IF;

    RETURN CONCAT('Sum: ', v_sum, ' | Average: ', ROUND(v_sum / v_count, 2));
END //

DELIMITER ;
